	<form id="rf">
		<fieldset id="fst">
			<legend>Select an option:</legend>
			<label><input type="radio" value="boy" class=0> boy</label><br>
			<label><input type="radio" value="girl" class=1> girl</label>
		</fieldset>
	</form>
	<script>
		const i=x=>document.getElementById(x),
			e=x=>document.createElement(x),
			c=(x,y)=>x.appendChild(y),
			msg1="if a binary bit can flip, a hundred bits to build this space",
			msg2 = "what expanse can be shaped by another choice, given the chance",
			bes = document.getElementsByTagName('body')[0].style,
			rf=i("rf");
		var f=1,c1=0,lfs=i("fst");
		function r(str,m,h) {
			let a=[...str]
			for (const [i,c] of a.entries()) {a[i]=(Math.random()<h)?m(i):c;}
			return a.join("");
		}
		document.addEventListener("DOMContentLoaded", () => {
			rf.addEventListener("change", (event) => {
				const o=event.target.value;
				c1 += parseInt(event.target.getAttribute("class"));

				if (o) {
					event.target.parentElement.style="font-weight:bold;font-style:italic;background-color:yellow;color:#000;";
					var p=(c1/f)*100,
						no1=`${o}-boy`,
						no2=`${o}-girl`,
						va1=no1,
						va2=no2;
					bes.backgroundColor='hsl(0 0 L%)'.replace(/L/,p);
					bes.color=(p>70)?'#000':'#fff';
					const fs=e("fieldset"),l=e("legend"),opt1=e("label"),opt2=e("label"),h=f/50;
					l.textContent="Thank you. Please select an option:";
					c(fs,l);
					(f>7)&&(no1=r(no1,(i)=>msg1[i],h),no2=r(no2,(i)=>msg2[i],h));
					(f>10)&&(l.textContent=r(l.textContent,(i)=>" ",h));
					opt1.innerHTML=`<input type="radio" value="${va1}" class=0> ${no1}<br>`;
					opt2.innerHTML=`<input type="radio" value="${va2}" class=1> ${no2}<br>`;
					c(fs,opt1);
					c(fs,opt2);
					c(rf,fs);
					lfs.disabled=true;
					window.scrollTo(0, document.body.scrollHeight);
					(f>50)&&(fs.disabled=true);
					lfs=fs;
					f++;
				}
			});
		});
	</script>
